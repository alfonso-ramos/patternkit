class Logger:
    _instance = None

    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
        return cls._instance

    def log(self, message: str) -> None:
        print(f"[log] {message}")


def use_logger() -> None:
    logger = Logger()
    logger.log("Mensaje emitido desde el Singleton en Python")

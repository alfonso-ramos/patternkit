#include <stdio.h>

void print_array(int *arr, int length) {
    for (int i = 0; i < length; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");
}

int main(void) {
    int numbers[] = {3, 5, 8, 13};
    int size = sizeof(numbers) / sizeof(numbers[0]);
    numbers[size - 1] = 21;
    print_array(numbers, size);
    return 0;
}
